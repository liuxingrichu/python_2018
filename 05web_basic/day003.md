### html标签：分节标签、html5标志性标签 ###

	<header>页眉</header>
	<hgroup>大纲显示多个分组</hgroup>
	<!-- 导航或菜单 -->
	<nav>html5语言标志性标签
	    <h5>header</h5>
	    <h5>nav</h5>
	    <h5>section</h5>
	    <h5>article</h5>
	    <h5>aside</h5>
	    <h5>footer</h5>
	</nav>
	<section>主题或概念</section>
	<section>
	    <h1>标题1</h1>
	    <h3>标题3</h3>
	</section>
	<section>
	    <h6>标题6</h6>
	    <article>独立内容</article>
	    <aside>侧边栏</aside>
	    <address>联系信息</address>
	    <detail>
	        <summary>概述</summary>
	        可隐藏或显示的详细信息内容
	    </detail>
	</section>
	<footer>页脚</footer>
